// Listen for SW SKIP_WAITING
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.addEventListener('message', e => {
    if (e.data.type==='SKIP_WAITING') {
      const b=document.createElement('div');
      b.style=`position:fixed;top:0;left:0;right:0;background:#4ecdc4;color:#000;padding:10px;text-align:center;z-index:10000`;
      b.innerHTML='New version!';
      document.body.append(b);
    }
  });
}
